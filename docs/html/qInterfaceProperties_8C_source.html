00039 // Correction for the boundary condition on the unit normal nHat on
00040 // walls to produce the correct contact angle.
00041 
00042 // The dynamic contact angle is calculated from the component of the
00043 // velocity on the direction of the interface, parallel to the wall.

00045 void Foam::qInterfaceProperties::correctContactAngle
00046 (
00047     surfaceVectorField::Boundary& nHatb,
00048     const surfaceVectorField::Boundary& gradAlphaf
00049 ) const
00050 {
00051     const fvMesh& mesh = alpha1_.mesh();
00052     const volScalarField::Boundary& abf = alpha1_.boundaryField();
00053 
00054     const fvBoundaryMesh& boundary = mesh.boundary();
00055 
00056     forAll(boundary, patchi)
00057     {
00058         if (isA<alphaContactAngleTwoPhaseFvPatchScalarField>(abf[patchi]))
00059         {
00060             alphaContactAngleTwoPhaseFvPatchScalarField& acap =
00061                 const_cast<alphaContactAngleTwoPhaseFvPatchScalarField&>
00062                 (
00063                     refCast<const alphaContactAngleTwoPhaseFvPatchScalarField>
00064                     (
00065                         abf[patchi]
00066                     )
00067                 );
00068 
00069             fvsPatchVectorField& nHatp = nHatb[patchi];
00070             const scalarField theta
00071             (
00072                 degToRad() * acap.theta(U_.boundaryField()[patchi], nHatp)
00073             );
00074 
00075             const vectorField nf
00076             (
00077                 boundary[patchi].nf()
00078             );
00079 
00080             // Reset nHatp to correspond to the contact angle
00081 
00082             const scalarField a12(nHatp & nf);
00083             const scalarField b1(cos(theta));
00084 
00085             scalarField b2(nHatp.size());
00086             forAll(b2, facei)
00087             {
00088                 b2[facei] = cos(acos(a12[facei]) - theta[facei]);
00089             }
00090 
00091             const scalarField det(1.0 - a12*a12);
00092 
00093             scalarField a((b1 - a12*b2)/det);
00094             scalarField b((b2 - a12*b1)/det);
00095 
00096             nHatp = a*nf + b*nHatp;
00097             nHatp /= (mag(nHatp) + deltaN_.value());
00098 
00099             acap.gradient() = (nf & nHatp)*mag(gradAlphaf[patchi]);
00100             acap.evaluate();
00101         }
00102     }
00103 }
