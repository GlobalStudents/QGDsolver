00173 void Foam::cosVelocityFvPatchVectorField::setPatchVelocities (vectorField& pV)
00174 {
00175     scalarField z = (this->patch().Cf() & Hdirection_) - minZ_;
00176     
00177     forAll(pV, iFace)
00178     {
00179     scalar t = this->patch().boundaryMesh().mesh().time().value();
00180     this->refGrad()[iFace] = vector (0.0, 0.0, 0.0);
00181     this->valueFraction()[iFace] = 1.0;
00182     pV[iFace] = 
00183         A_ * cos(Foam::constant::mathematical::pi * z[iFace] / H_) * (-omega0_) * sin(omega0_ * t + phi0_);
00184     if (z[iFace] < 0.0 || z[iFace] > H_)
00185     {
00186         pV[iFace] = A_*0.0;
